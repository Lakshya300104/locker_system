To create the database:

        CREATE DATABASE locker_system;
        USE locker_system;

(Make sure you use the same name for database, or change the database name in the DBConnection file)

user TABLE:
        CREATE TABLE users (
            user_id INT AUTO_INCREMENT PRIMARY KEY,
            username VARCHAR(50) NOT NULL UNIQUE,
            password VARCHAR(100) NOT NULL,
            role VARCHAR(10) NOT NULL,      -- 'student' or 'admin'
            full_name VARCHAR(100)
        );

locker TABLE:

        CREATE TABLE lockers (
            locker_id INT PRIMARY KEY,
            status VARCHAR(20) DEFAULT 'available',
            assigned_to INT NULL,
            locker_password VARCHAR(20),
            FOREIGN KEY (assigned_to) REFERENCES users(user_id)
        );
